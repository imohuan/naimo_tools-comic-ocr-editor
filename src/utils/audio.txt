import React, { useState, useRef, useEffect } from 'react';
import { Upload, Play, FileAudio, Music, X, Download, Loader2, RefreshCw } from 'lucide-react';

const AudioMerger = () => {
  const [files, setFiles] = useState([]);
  const [isProcessing, setIsProcessing] = useState(false);
  const [progress, setProgress] = useState('');
  const [mergedAudioUrl, setMergedAudioUrl] = useState(null);
  const audioContextRef = useRef(null);

  // 初始化 AudioContext
  useEffect(() => {
    const AudioContext = window.AudioContext || window.webkitAudioContext;
    audioContextRef.current = new AudioContext();
    return () => {
      if (audioContextRef.current) {
        audioContextRef.current.close();
      }
    };
  }, []);

  const handleFileChange = (e) => {
    const selectedFiles = Array.from(e.target.files);
    // 简单的去重，基于文件名
    const newFiles = selectedFiles.filter(
      (file) => !files.some((f) => f.name === file.name)
    );
    setFiles((prev) => [...prev, ...newFiles]);
  };

  const removeFile = (index) => {
    setFiles((prev) => prev.filter((_, i) => i !== index));
    if (mergedAudioUrl) {
      URL.revokeObjectURL(mergedAudioUrl);
      setMergedAudioUrl(null);
    }
  };

  const moveFile = (index, direction) => {
    const newFiles = [...files];
    if (direction === -1 && index > 0) {
      [newFiles[index], newFiles[index - 1]] = [newFiles[index - 1], newFiles[index]];
    } else if (direction === 1 && index < newFiles.length - 1) {
      [newFiles[index], newFiles[index + 1]] = [newFiles[index + 1], newFiles[index]];
    }
    setFiles(newFiles);
    setMergedAudioUrl(null); // 顺序改变需重新合并
  };

  // 核心功能：将 AudioBuffer 转换为 WAV Blob
  const bufferToWave = (abuffer, len) => {
    const numOfChan = abuffer.numberOfChannels;
    const length = len * numOfChan * 2 + 44;
    const buffer = new ArrayBuffer(length);
    const view = new DataView(buffer);
    const channels = [];
    let i;
    let sample;
    let offset = 0;
    let pos = 0;

    // 写入 WAV 头部
    setUint32(0x46464952); // "RIFF"
    setUint32(length - 8); // file length - 8
    setUint32(0x45564157); // "WAVE"

    setUint32(0x20746d66); // "fmt " chunk
    setUint32(16); // length = 16
    setUint16(1); // PCM (uncompressed)
    setUint16(numOfChan);
    setUint32(abuffer.sampleRate);
    setUint32(abuffer.sampleRate * 2 * numOfChan); // avg. bytes/sec
    setUint16(numOfChan * 2); // block-align
    setUint16(16); // 16-bit (hardcoded in this example)

    setUint32(0x61746164); // "data" - chunk
    setUint32(length - pos - 4); // chunk length

    // 写入交错数据 (Interleaved PCM)
    for (i = 0; i < numOfChan; i++)
      channels.push(abuffer.getChannelData(i));

    while (pos < len) {
      for (i = 0; i < numOfChan; i++) {
        // clamp float32 to int16 range
        sample = Math.max(-1, Math.min(1, channels[i][pos]));
        // convert to 16-bit signed integer
        sample = (0.5 + sample < 0 ? sample * 32768 : sample * 32767) | 0;
        view.setInt16(44 + offset, sample, true);
        offset += 2;
      }
      pos++;
    }

    return new Blob([buffer], { type: "audio/wav" });

    function setUint16(data) {
      view.setUint16(pos, data, true);
      pos += 2;
    }

    function setUint32(data) {
      view.setUint32(pos, data, true);
      pos += 4;
    }
  };

  const mergeAudio = async () => {
    if (files.length === 0) return;
    setIsProcessing(true);
    setProgress('正在初始化...');
    
    try {
      const ctx = audioContextRef.current;
      const audioBuffers = [];

      // 1. 解码所有文件
      for (let i = 0; i < files.length; i++) {
        setProgress(`正在解码第 ${i + 1} / ${files.length} 个文件...`);
        const arrayBuffer = await files[i].arrayBuffer();
        // create a copy because decodeAudioData detaches the buffer
        const decodedBuffer = await ctx.decodeAudioData(arrayBuffer);
        audioBuffers.push(decodedBuffer);
      }

      setProgress('正在合并音频数据...');

      // 2. 计算总长度和通道数
      const outputChannels = 2; // 强制双声道以保证兼容性
      let totalLength = 0;
      audioBuffers.forEach(buf => totalLength += buf.length);

      // 3. 创建输出 Buffer
      const outputBuffer = ctx.createBuffer(
        outputChannels,
        totalLength,
        ctx.sampleRate
      );

      // 4. 将数据拷贝到输出 Buffer (拼接)
      let offset = 0;
      audioBuffers.forEach((buf) => {
        for (let channel = 0; channel < outputChannels; channel++) {
          // 获取输出通道的数据引用
          const outputData = outputBuffer.getChannelData(channel);
          
          // 如果源文件通道少于当前需要的通道 (例如单声道源 -> 双声道输出)
          // 我们复用单声道数据。如果源文件有多通道，直接对应。
          const inputData = buf.getChannelData(
            channel < buf.numberOfChannels ? channel : 0
          );

          // 高效拷贝
          outputData.set(inputData, offset);
        }
        offset += buf.length;
      });

      setProgress('正在封装为 WAV 文件...');
      
      // 5. 编码为 WAV Blob (为了生成可拖拽进度的 URL)
      // 注意：这里是一个同步阻塞操作，非常大的文件可能会卡顿一下
      // 在生产环境中通常放在 Web Worker 中处理
      setTimeout(() => {
        try {
            const wavBlob = bufferToWave(outputBuffer, totalLength);
            const url = URL.createObjectURL(wavBlob);
            setMergedAudioUrl(url);
            setIsProcessing(false);
            setProgress('完成！');
        } catch (err) {
            console.error(err);
            setProgress('封装失败');
            setIsProcessing(false);
        }
      }, 100); // 稍微延迟以允许 UI 更新

    } catch (error) {
      console.error("Merge error:", error);
      setProgress('处理出错: ' + error.message);
      setIsProcessing(false);
    }
  };

  const formatFileSize = (bytes) => {
    if (bytes === 0) return '0 B';
    const k = 1024;
    const sizes = ['B', 'KB', 'MB', 'GB'];
    const i = Math.floor(Math.log(bytes) / Math.log(k));
    return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];
  };

  return (
    <div className="min-h-screen bg-slate-900 text-slate-100 p-6 font-sans">
      <div className="max-w-3xl mx-auto space-y-8">
        
        {/* Header */}
        <div className="text-center space-y-2">
          <div className="flex justify-center mb-4">
            <div className="p-4 bg-indigo-600 rounded-full shadow-lg shadow-indigo-500/30">
              <Music size={40} className="text-white" />
            </div>
          </div>
          <h1 className="text-3xl font-bold bg-clip-text text-transparent bg-gradient-to-r from-indigo-400 to-cyan-400">
            无缝音频合并神器
          </h1>
          <p className="text-slate-400">
            上传多个音频，合并为一个可随意跳转的独立文件。
            <br/>
            <span className="text-xs text-slate-500">纯前端处理，数据不上传服务器</span>
          </p>
        </div>

        {/* Upload Area */}
        <div className="bg-slate-800 rounded-2xl p-8 border-2 border-dashed border-slate-600 hover:border-indigo-500 transition-colors cursor-pointer relative group">
          <input
            type="file"
            multiple
            accept="audio/*"
            onChange={handleFileChange}
            className="absolute inset-0 w-full h-full opacity-0 cursor-pointer z-10"
          />
          <div className="flex flex-col items-center justify-center space-y-4 text-slate-400 group-hover:text-indigo-400 transition-colors">
            <Upload size={48} />
            <div className="text-lg font-medium">点击或拖拽音频文件到此处</div>
            <div className="text-sm">支持 MP3, WAV, AAC, OGG 等</div>
          </div>
        </div>

        {/* File List */}
        {files.length > 0 && (
          <div className="bg-slate-800 rounded-2xl p-6 shadow-xl">
            <div className="flex justify-between items-center mb-4">
              <h2 className="text-xl font-semibold flex items-center gap-2">
                <FileAudio className="text-indigo-400" />
                待合并列表 ({files.length})
              </h2>
              <button 
                onClick={() => { setFiles([]); setMergedAudioUrl(null); }}
                className="text-xs text-red-400 hover:text-red-300"
              >
                清空列表
              </button>
            </div>
            
            <div className="space-y-2 max-h-60 overflow-y-auto pr-2 custom-scrollbar">
              {files.map((file, index) => (
                <div key={`${file.name}-${index}`} className="flex items-center justify-between bg-slate-700/50 p-3 rounded-lg border border-slate-700 hover:border-indigo-500/50 transition-all">
                  <div className="flex items-center gap-3 overflow-hidden">
                    <span className="bg-slate-600 text-xs w-6 h-6 flex items-center justify-center rounded-full shrink-0">
                      {index + 1}
                    </span>
                    <div className="min-w-0">
                      <p className="truncate font-medium text-sm text-slate-200">{file.name}</p>
                      <p className="text-xs text-slate-500">{formatFileSize(file.size)}</p>
                    </div>
                  </div>
                  
                  <div className="flex items-center gap-2 shrink-0">
                    <div className="flex flex-col gap-1 mr-2">
                      <button 
                        onClick={() => moveFile(index, -1)} 
                        disabled={index === 0}
                        className="p-1 hover:bg-slate-600 rounded disabled:opacity-30"
                      >
                        <div className="w-0 h-0 border-l-[4px] border-l-transparent border-r-[4px] border-r-transparent border-b-[6px] border-b-slate-300"></div>
                      </button>
                      <button 
                        onClick={() => moveFile(index, 1)} 
                        disabled={index === files.length - 1}
                        className="p-1 hover:bg-slate-600 rounded disabled:opacity-30"
                      >
                        <div className="w-0 h-0 border-l-[4px] border-l-transparent border-r-[4px] border-r-transparent border-t-[6px] border-t-slate-300"></div>
                      </button>
                    </div>
                    <button
                      onClick={() => removeFile(index)}
                      className="p-2 text-slate-400 hover:text-red-400 hover:bg-red-400/10 rounded-full transition-colors"
                    >
                      <X size={18} />
                    </button>
                  </div>
                </div>
              ))}
            </div>

            {/* Merge Button */}
            <div className="mt-6">
              <button
                onClick={mergeAudio}
                disabled={isProcessing}
                className={`w-full py-4 rounded-xl font-bold text-lg shadow-lg flex items-center justify-center gap-2 transition-all ${
                  isProcessing 
                    ? 'bg-slate-600 cursor-not-allowed text-slate-400'
                    : 'bg-indigo-600 hover:bg-indigo-500 text-white hover:shadow-indigo-500/25 active:scale-[0.99]'
                }`}
              >
                {isProcessing ? (
                  <>
                    <Loader2 className="animate-spin" />
                    {progress}
                  </>
                ) : (
                  <>
                    <RefreshCw />
                    开始合并处理
                  </>
                )}
              </button>
            </div>
          </div>
        )}

        {/* Result Area */}
        {mergedAudioUrl && (
          <div className="bg-gradient-to-br from-indigo-900/50 to-slate-800 rounded-2xl p-8 border border-indigo-500/30 shadow-2xl animation-fade-in">
            <h3 className="text-2xl font-bold mb-6 text-center text-indigo-200">合并完成！</h3>
            
            <div className="bg-slate-900/50 p-4 rounded-xl border border-indigo-500/20 mb-6">
              <audio 
                controls 
                className="w-full h-12 outline-none" 
                src={mergedAudioUrl}
              >
                您的浏览器不支持 audio 元素。
              </audio>
            </div>

            <div className="flex justify-center">
              <a
                href={mergedAudioUrl}
                download={`merged-audio-${new Date().getTime()}.wav`}
                className="flex items-center gap-2 px-6 py-3 bg-emerald-600 hover:bg-emerald-500 text-white rounded-lg font-semibold transition-colors shadow-lg shadow-emerald-900/20"
              >
                <Download size={20} />
                下载合并后的文件 (WAV)
              </a>
            </div>
          </div>
        )}

      </div>
      
      <style>{`
        .custom-scrollbar::-webkit-scrollbar {
          width: 6px;
        }
        .custom-scrollbar::-webkit-scrollbar-track {
          background: rgba(255, 255, 255, 0.05);
          border-radius: 4px;
        }
        .custom-scrollbar::-webkit-scrollbar-thumb {
          background: rgba(255, 255, 255, 0.2);
          border-radius: 4px;
        }
        .custom-scrollbar::-webkit-scrollbar-thumb:hover {
          background: rgba(255, 255, 255, 0.3);
        }
        @keyframes fadeIn {
          from { opacity: 0; transform: translateY(10px); }
          to { opacity: 1; transform: translateY(0); }
        }
        .animation-fade-in {
          animation: fadeIn 0.5s ease-out forwards;
        }
      `}</style>
    </div>
  );
};
